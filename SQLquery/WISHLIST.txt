variant 1

CREATE TABLE wishlist (
id INT UNSIGNED NOT NULL AUTO_INCREMENT,
PRIMARY KEY (id),
time TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
name VARCHAR(100) NOT NULL,
cost DECIMAL(9,3),
status VARCHAR(20),
info VARCHAR(200)
) ENGINE = InnoDB;
